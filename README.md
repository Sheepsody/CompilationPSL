## Syntaxe d'un programme

```shell
/*
Commentaires

# Variables
let a = 3;
let b = 3;
a = b = 1;

# Function
fn (a, b) {
    let a = 5;
    5 # Retour implicite sans ;
}

# If-Else condition
if cond
then {
    ... ;
    ... ;
} else {
    ... ;
    ... ;
}

# While loop
while cond {
    ... ;
}

# Global variables
global a = 3;
fn test () {
    a
}

5 # Retour implicite sans ;
```
